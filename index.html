def verificar_superposicion(orto_bounds, shape_bounds):
    """
    Verifica si hay superposicin entre la ortofoto y los polgonos
    """
    orto_minx, orto_miny, orto_maxx, orto_maxy = orto_bounds
    
    for idx, row in shape_bounds.iterrows():
        shape_minx, shape_miny, shape_maxx, shape_maxy = row[['minx', 'miny', 'maxx', 'maxy']]
        
        # Verificar superposicin
        overlap_x = max(0, min(orto_maxx, shape_maxx) - max(orto_minx, shape_minx))
        overlap_y = max(0, min(orto_maxy, shape_maxy) - max(orto_miny, shape_miny))
        
        if overlap_x > 0 and overlap_y > 0:
            print(f"Polgono {idx} S se superpone con la ortofoto")
            return True
        else:
            print(f"Polgono {idx} NO se superpone con la ortofoto")
            print(f"  Ortofoto: ({orto_minx:.2f}, {orto_miny:.2f}) - ({orto_maxx:.2f}, {orto_maxy:.2f})")
            print(f"  Polgono: ({shape_minx:.2f}, {shape_miny:.2f}) - ({shape_maxx:.2f}, {shape_maxy:.2f})")
    
    return False

def crear_mascara_valida(img_recortada, buffer_pixels=10):
    """
    Crea mscara vlida con verificacin de tamao
    """
    h, w = img_recortada.shape
    print(f"Creando mscara vlida para imagen de {h}x{w} pxeles")
    
    # Ajustar buffer si es muy grande para la imagen
    max_buffer = min(h//4, w//4, buffer_pixels)
    if max_buffer < buffer_pixels:
        print(f"Ajustando buffer de {buffer_pixels} a {max_buffer} pxeles")
        buffer_pixels = max_buffer
    
    # Si la imagen es muy pequea, usar buffer mnimo
    if h <= 20 or w <= 20:
        buffer_pixels = max(1, min(2, buffer_pixels))
        print(f"Imagen muy pequea, usando buffer de {buffer_pixels} pxeles")
    
    mask = np.ones(img_recortada.shape, dtype=bool)
    
    if buffer_pixels > 0:
        mask[:buffer_pixels, :] = False  # borde superior
        mask[-buffer_pixels:, :] = False  # borde inferior
        mask[:, :buffer_pixels] = False  # borde izquierdo
        mask[:, -buffer_pixels:] = False  # borde derecho
    
    pixels_validos = np.sum(mask)
    print(f"Pxeles vlidos despus del buffer: {pixels_validos}")
    
    return mask, buffer_pixels

def limpiar_bordes_morfologia(mascara_grietas, gray_img, min_size=10):
    """
    Limpieza morfolgica con parmetros adaptativos
    """
    # Ajustar min_size segn el tamao de la imagen
    total_pixels = gray_img.shape[0] * gray_img.shape[1]
    min_size = max(min_size, total_pixels // 1000)  # Al menos 0.1% del total
    
    kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (3, 3))
    cleaned = cv2.morphologyEx(mascara_grietas.astype(np.uint8), cv2.MORPH_OPEN, kernel)
    cleaned_bool = cleaned.astype(bool)
    
    # Solo aplicar remove_small_objects si hay objetos suficientes
    if np.sum(cleaned_bool) > min_size:
        cleaned_bool = remove_small_objects(cleaned_bool, min_size=min_size)
    
    return cleaned_bool.astype(np.uint8)

# Convertir mscara a polgonos usando rasterio.features.shapes
def crear_poligonos_desde_mascara(mascara, transform):
    """
    Convierte una mscara binaria a polgonos georreferenciados
    """
    polygons = []
    
    # Extraer formas de la mscara
    for geom, value in shapes(mascara.astype(np.uint8), transform=transform):
        if value == 1:  # Solo los pxeles de grietas (valor 1)
            polygons.append(Polygon(geom['coordinates'][0]))
    
    return polygons
